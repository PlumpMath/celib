  28 /* The true or false value for CeBool */
  29 #define CE_FALSE 0
  30 #define CE_TRUE  1
  31 
  32 /* Convert str to string */
  33 #define _CE_STR(str) #str
  34 #define CE_STR(str) _CC_STR(str)

  36 /* Find Max value or Min value */
  37 #define CE_MAX(x,y) ( (x) > (y) ? (x) : (y) )
  38 #define CE_MIN(x,y) ( (x) < (y) ? (x) : (y) )
  39 
  40 #define CE_POWER_2(x) ( (x) * (x) )
  41 #define CE_POWER_3(x) ( (x) * (x) * (x) )
  42 
  43 /* Make a character to upcase or downcase */
  44 #define CE_UPCASE(c)   ( ( (c) >= 'a' && (c) <= 'z') ? ( (c) - 0x20) : (c) )
  45 #define CE_DOWNCASE(c) ( ( (c) >= 'A' && (c) <= 'Z') ? ( (c) + 0x20) : (c) )
  46 
  47 /* Shift the value to left or right */
  48 #define CE_LSHIFT(x, offset) ( (x) << (offset) )
  49 #define CE_RSHIFT(x, offset) ( (x) >> (offset) )
  50 
  51 /* Logic Arithmatic */
  52 #define CE_NOT(a)    ( ~(a) )
  53 #define CE_OR(a, b)  ( (a) | (b) )
  54 #define CE_AND(a, b) ( (a) & (b) )
  55 #define CE_XOR(a, b) ( (a) ^ (b) )
  56 #define CE_NOR(a, b)  ( ~( (a) | (b) ) )
  57 #define CE_NAND(a, b) ( ~( (a) & (b) ) )
  58 
  59 /* Initial the range */
  60 #define CE_RANGE_INITIAL(start, end, length)                          \
  61                 do {                                                          \
  62                                 /* Reset start and end variables */                   \
  63                                 start += ( start > 0 ) ? (-1) : (length);             \
  64                                 end   += ( end   > 0 ) ? (-1) : (length);             \
  65                                 /* If start gratter than end, we need to swap them */ \
  66                                 if ( start > end ) {                                  \
  67                                                 ce_int_swap(&start, &end);                    \
  68                                 }                                                     \
  69                 } while(0)
